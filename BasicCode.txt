#!/bin/bash

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

TARGET=$1
OUTPUT_DIR="bounty_$TARGET"

# Create output directory
mkdir -p $OUTPUT_DIR
cd $OUTPUT_DIR

echo -e "${GREEN}[+] Starting Bug Bounty Automation for $TARGET${NC}"

# PHASE 1: Subdomain Discovery
echo -e "${YELLOW}[1/11] Subdomain Discovery...${NC}"
subfinder -d $TARGET -all -o subfinder.txt
assetfinder --subs-only $TARGET > assetfinder.txt
amass enum -passive -d $TARGET -o amass.txt
cat subfinder.txt assetfinder.txt amass.txt | sort -u > all_subs.txt
echo -e "${GREEN}[+] Found $(cat all_subs.txt | wc -l) subdomains${NC}"

# PHASE 2: URL Discovery
echo -e "${YELLOW}[2/11] URL Discovery...${NC}"
cat all_subs.txt | gau > gau_urls.txt
cat all_subs.txt | waybackurls > wayback_urls.txt
cat gau_urls.txt wayback_urls.txt | sort -u > all_urls.txt
echo -e "${GREEN}[+] Collected $(cat all_urls.txt | wc -l) URLs${NC}"

# PHASE 3: Live Hosts Probing
echo -e "${YELLOW}[3/11] Probing Live Hosts...${NC}"
cat all_subs.txt | httpx -silent -status-code -title -o live_hosts.txt
echo -e "${GREEN}[+] Found $(cat live_hosts.txt | wc -l) live hosts${NC}"

# PHASE 4: Parameter Discovery
echo -e "${YELLOW}[4/11] Extracting Parameters...${NC}"
cat all_urls.txt | grep "?" | grep "=" > params_urls.txt
echo -e "${GREEN}[+] Found $(cat params_urls.txt | wc -l) URLs with parameters${NC}"

# PHASE 5: Directory Fuzzing
echo -e "${YELLOW}[5/11] Directory Fuzzing...${NC}"
ffuf -w /usr/share/wordlists/dirb/common.txt -u https://$TARGET/FUZZ -o ffuf.json -of json
echo -e "${GREEN}[+] Directory fuzzing completed${NC}"

# PHASE 6: Port Scanning
echo -e "${YELLOW}[6/11] Port Scanning...${NC}"
nmap -sV -sC -T4 -oA nmap_scan $TARGET
echo -e "${GREEN}[+] Port scan completed${NC}"

# PHASE 7: Vulnerability Scanning
echo -e "${YELLOW}[7/11] Vulnerability Scanning...${NC}"
cat live_hosts.txt | nuclei -t ~/nuclei-templates/ -severity critical,high -o nuclei_vulns.txt
echo -e "${GREEN}[+] Found $(cat nuclei_vulns.txt | wc -l) critical/high vulnerabilities${NC}"

# PHASE 8: SQL Injection Testing
echo -e "${YELLOW}[8/11] SQL Injection Testing...${NC}"
sqlmap -m params_urls.txt --batch --level 3 --risk 2 -o sqlmap_results.txt
echo -e "${GREEN}[+] SQL injection testing completed${NC}"

# PHASE 9: XSS Testing
echo -e "${YELLOW}[9/11] XSS Testing...${NC}"
dalfox file params_urls.txt --only-poc -o dalfox_xss.txt
echo -e "${GREEN}[+] XSS testing completed${NC}"

# PHASE 10: OSINT Gathering
echo -e "${YELLOW}[10/11] OSINT Gathering...${NC}"
theHarvester -d $TARGET -b all -f theharvester.html
whois $TARGET > whois.txt
echo -e "${GREEN}[+] OSINT gathering completed${NC}"

# PHASE 11: Report Generation
echo -e "${YELLOW}[11/11] Generating Report...${NC}"
cat > report.txt << EOF
BUG BOUNTY REPORT FOR $TARGET
================================
Generated: $(date)

SUMMARY:
- Subdomains Found: $(cat all_subs.txt | wc -l)
- Live Hosts: $(cat live_hosts.txt | wc -l)
- Total URLs: $(cat all_urls.txt | wc -l)
- URLs with Parameters: $(cat params_urls.txt | wc -l)
- Critical/High Vulns: $(cat nuclei_vulns.txt | wc -l)

FILES GENERATED:
1. all_subs.txt - All subdomains
2. live_hosts.txt - Live hosts
3. all_urls.txt - All URLs
4. params_urls.txt - URLs with parameters
5. nuclei_vulns.txt - Vulnerability findings
6. sqlmap_results.txt - SQL injection results
7. dalfox_xss.txt - XSS findings
8. nmap_scan.xml - Port scan results
9. theharvester.html - OSINT data
10. whois.txt - Domain registration info

Check individual files for detailed results.
EOF

echo -e "${GREEN}[+] Report saved to report.txt${NC}"
echo -e "${GREEN}[+] All results saved in $OUTPUT_DIR/${NC}"

# Final summary
echo -e "\n${GREEN}========================================${NC}"
echo -e "${GREEN}SCAN COMPLETED SUCCESSFULLY!${NC}"
echo -e "${GREEN}========================================${NC}"
echo -e "Total time: $(date)"
